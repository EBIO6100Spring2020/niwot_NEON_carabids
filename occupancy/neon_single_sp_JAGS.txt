model { 

# Priors 
psi ~ dunif(0, 1) 
p ~ dunif(0, 1) 

# Likelihood 
# Ecological model for true occurrence 
for (i in 1:R) { 
    z[i] ~ dbern(psi) 
    p.eff[i] <- z[i] * p 
    # Observation model for replicated detection/nondetection observations
    for (j in 1:T) { 
        y[i,j] ~ dbern(p.eff[i])
    } #j 
} #i
# Derived quantities 
occ.fs <- sum(z[])

} #model
