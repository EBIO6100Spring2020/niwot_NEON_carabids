model {
  # priors
  psi_beta1 ~ dunif(0, 20)
  psi_beta2 ~ dunif(0, 20)
  p_beta1 ~ dunif(0, 20)
  p_beta2 ~ dunif(0, 20)

  for (i in 1:nspec) {
    psi[i] ~ dbeta(psi_beta1, psi_beta2)
    p[i] ~ dbeta(p_beta1, p_beta2)
  }
  
  # state model
  for (j in 1:R) {
    for (i in 1:nspec) {
      Z[j, i]  ~ dbern(psi[i])
    }
  }

  # observation model
  for (j in 1:R) {
    for (i in 1:nspec) {
      Y[j, i] ~ dbinom(Z[j, i] * p[i], T)
    }
  }
}
